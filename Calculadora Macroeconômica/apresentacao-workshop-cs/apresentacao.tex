\begin{frame}{A dor: dados prontos, apresentação represada}
  \takeaway{O gargalo de uma apresentação raramente é a análise, é transformar a análise em slides consistentes e bem desenhados na velocidade que o negócio pede.}
  \fonte{referência interna A\&M, estrutura-deck.md}
  \vspace{4mm}
  \begin{columns}[T, onlytextwidth]
    \begin{column}{0.58\textwidth}
      {\bfseries Sintomas comuns}\par\vspace{1.5mm}
      \begin{itemize}
        \item Horas redistribuindo a mesma tabela em formatos diferentes a cada apresentação
        \item Gráficos refeitos manualmente sempre que um dado é atualizado
        \item Identidade visual inconsistente entre decks do mesmo projeto
        \item Revisão de conteúdo e revisão de design acontecendo nas mesmas idas e voltas
      \end{itemize}
    \end{column}
    \begin{column}{0.38\textwidth}
      \amhighlight{8--12}{slides por proposta A\&M, hoje gastos em formatação}
      \vspace{4mm}
      \begin{amcard}
        {\bfseries Custo real}\par\vspace{1.5mm}
        Cada deck consome ciclos de analista sênior em tarefa de formatação, não de julgamento.\par\vspace{2mm}
        É o intervalo que, hoje, amarra contexto, escopo e investimento em cada proposta.
      \end{amcard}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Como funciona o apresentacao-rapida-am}
  \takeaway{Dividir a criação do deck em sete etapas com validação cruzada entre agentes é o que permite trocar reunião de design por revisão automática, sem perder rigor.}
  \vspace{5mm}
  \begin{center}
    \includegraphics[width=0.96\textwidth]{decoracao/_valida_d1.pdf}
  \end{center}
  \vspace{4mm}
  \begin{columns}[T, onlytextwidth]
    \begin{column}{0.5\textwidth}
      {\bfseries O que isso garante}\par\vspace{1mm}
      Cada uma das 7 etapas, do roteiro ao código LaTeX, passa por validação própria antes de avançar para a próxima, o que elimina retrabalho de formatação e inconsistência visual entre seções do mesmo deck.
    \end{column}
    \begin{column}{0.5\textwidth}
      {\bfseries O que isso elimina}\par\vspace{1mm}
      \begin{itemize}
        \item Retrabalho manual de formatação e diagramação
        \item Inconsistência visual entre seções do mesmo deck
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{A metodologia que sustenta cada número}
  \takeaway{Cada valor corrigido combina dados realizados em fator acumulado com projeções de mercado, sempre identificando qual parte do cálculo é fato e qual é expectativa.}
  \fonte{ENGINE\_CONTEXT.md, API SGS e Boletim Focus do Banco Central do Brasil (2026)}
  \vspace{4mm}
  \begin{columns}[T, onlytextwidth]
    \begin{column}{0.58\textwidth}
      \begin{tabular}{lccc}
        \rowcolor{amNavy}\color{white}Índice & \color{white}Fonte SGS & \color{white}Início & \color{white}Projeção \\
        \rowcolor{amTableStripe}IPCA & 433 & 07/1994 & Focus + ETTJ \\
        IGP-M & 189 & 01/1940 & Focus \\
        \rowcolor{amTableStripe}IPCA-15 & 7478 & 02/1999 & Focus \\
        INPC & 188 & 01/1979 & Focus \\
        \rowcolor{amTableStripe}INCC & 192 & 01/1985 & -- \\
        SELIC & 4390 & 06/1986 & -- \\
        \rowcolor{amTableStripe}CDI & 4391 & 06/1986 & -- \\
      \end{tabular}
    \end{column}
    \begin{column}{0.38\textwidth}
      {\bfseries Como o fator é calculado}\par\vspace{1.5mm}
      {\bfseries\color{amSignal}factor = \(\prod (1+r_t/100)\)}\par\vspace{1.5mm}
      \begin{itemize}
        \item Realizado: produto encadeado das taxas mensais entre as duas datas
        \item Focus: taxa anual mediana distribuída em taxa mensal composta uniforme
        \item ETTJ: taxas forward da curva de inflação implícita ANBIMA (NTN-B), só para IPCA
      \end{itemize}
      \vspace{4mm}\textcolor{amMist}{\rule{0.92\linewidth}{0.3pt}}\vspace{3mm}
      {\amcondlightfont\color{amGray}IPCA, IGP-M, IPCA-15 e INPC têm projeção Focus, e o IPCA soma também a opção ETTJ. INCC, SELIC e CDI não projetam: a data final fica limitada ao último dado realizado.}
    \end{column}
  \end{columns}
\end{frame}
